    Qt Designer

    Pre-requisition: have PyQt installed in the virtual environment or in the project

    • PyQt5: use $ pip install pyqt5-tools
    • PyQt6: use $ pip install pyqt6-tools

    Open Qt Designer by using the command qt6-tools designer (or qt5-tools designer) in terminal. Noted that you can only open Qt Designer from terminal when working in a virtual environment.

    Another way to open Qt Designer, in your IDE, go to venv/lib/python3.8/site-packages/qt6_applications/Qt/bin. Open the folder and double-click the Designer.exe (on Windows) or Designer.app (on macOS)

    Note

    I find that QT Designer in this method not nice to working with. The GUI and toolbars aren't combined in a single window.

    (Optional) Qt Designer for Mac (Qt Creator)

    Download and Install

    • Go to download page.
    • Download open source
    • Create an account and verify (or later)
    • If you haven't Xcode, download it
    • Open the .dmg file
    • Install

    The installment could take an hour depending on your internet speed and the server download speed. On my side, it took almost 2 hours to download 5.32 Gb.

    Open and Create a new Qt project

    • Open Qt Designer
    • Go to File/New File.... This method will allow to create a .ui file for the project and save it to our project folder
    • In Files and CLasses select Qt then Qt Designer Form and click the Choose... button
    • In Choose a Form Template, select the template you want such as Main Window and click the Continue button
    • Enter a File name and the Path to save, and click the Continue button
    • Next, you can add version control (if any) and then click the Done button

    In my opinion, Qt Creator looks better than Qt Designer and run in parallel to the IDE, unlike the Qt Designer.

    Create Window

    Let's start a simple window with the size of 800x600 and a title of "First Window".

    # Import dependencies
from PyQt6.QtWidgets import QApplication, QWidget
from PyQt6.QtGui import *
import sys


class Window(QWidget):
    def __init__(self):
        super().__init__()

        self.setGeometry(200, 200, 800, 600)
        self.setWindowTitle("First Window")


# Initialize and loop the window
app = QApplication(sys.argv)
window = Window()
window.show()
sys.exit(app.exec())

    As of the code above, when we run the code we will have a single window app running and will be closed only when we hit the close button or terminate the code.

    Run Qt Creator

    To run the app with the created GUI from Qt Designer we need to import another module called uic.

    • In the main script, add the line, from PyQt6 import uic

    Events

    # Import dependencies
from PyQt6.QtWidgets import QApplication, QWidget, QLineEdit, QLabel, QHBoxLayout, QPushButton
from PyQt6.QtGui import QFont
import sys


class Window(QWidget):
    def __init__(self):
        """
        Initiate Window
        """
        super().__init__()

        self.setGeometry(200, 200, 800, 600)  # set window size
        self.setWindowTitle("Events")  # set window title

        self.create_widgets()  # call create_widget() to initialize the widgets in the window

    def create_widgets(self):
        """
        Create widgets
        """
        self.line_edit = QLineEdit(self)  # initiate text input
        self.line_edit.setPlaceholderText("Enter Text Here...")  # set default text

        button = QPushButton()  # initiate button
        button.clicked.connect(self.on_click_button)  # connect button to a method

        self.label = QLabel("Waiting for text...")  # initiate label

        box = QHBoxLayout()  # initiate horizontal alignment box
        box.addWidget(self.line_edit)  # add text input to alignment box
        box.addWidget(button)  # add button to alignment box
        box.addWidget(self.label)  # add label to alignment box

        self.setLayout(box)  # add alignment box to the window

    def on_click_button(self):
        """
        Change the text label according to the input box when click on the button
        """
        text = self.line_edit.text()  # get text from the text input box
        self.label.setText(text)  # set label text to the new text input


# Initialize and loop the window
app = QApplication(sys.argv)
window = Window()
window.show()
sys.exit(app.exec())

    QPushButton, QMenu

    from PyQt6.QtWidgets import QApplication, QWidget, QPushButton, QMenu
from PyQt6.QtGui import *
from PyQt6.QtCore import QSize
import sys


class Window(QWidget):
    def __init__(self):
        super().__init__()

        self.setGeometry(200, 200, 800, 600)
        self.setWindowTitle("First Window")

        self.create_button()

    def create_button(self):
        # Create the button
        button = QPushButton("Click Me!", self)  # initiate button (str: 'button_text', obj: parent)
        button.setGeometry(100, 100, 150, 150)  # set size and location  (int: w, int: h, int: x, int: y)
        button.setFont(QFont('Sansarif', 15, QFont.Weight.ExtraBold))  # set button font
        button.setIcon(QIcon('images/save.png'))  # set button icon
        button.setIconSize(QSize(50, 50))  # set button icon size

        # Drop down menu on the button
        menu = QMenu()  # initiate menu
        menu.setFont(QFont("Sansarif", 15, QFont.Weight.ExtraBold))
        menu.setStyleSheet("background-color: red")
        menu.addAction("Open...")
        menu.addAction("Cut")
        menu.addAction("Paste")
        button.setMenu(menu)  # attach menu to the button


app = QApplication(sys.argv)
window = Window()
window.show()
sys.exit(app.exec())

    QToolBars

    import sys
from PyQt6.QtCore import *
from PyQt6.QtGui import *
from PyQt6.QtWidgets import *


class Demo(QMainWindow):
    def __init__(self, parent=None):
        super(Demo, self).__init__(parent)
        layout = QVBoxLayout()
        tb = self.addToolBar("File")

        new = QAction(QIcon("images/new.png"), "new", self)
        tb.addAction(new)

        o = QAction(QIcon("images/open.png"), "open", self)
        tb.addAction(o)
        save = QAction(QIcon("images/save.png"), "save", self)
        tb.addAction(save)
        tb.actionTriggered[QAction].connect(self.on_click_button)
        self.setLayout(layout)
        self.setWindowTitle("toolbar demo")

    def on_click_button(self, a):
        print("pressed tool button is", a.text())


def main():
    app = QApplication(sys.argv)
    window = Demo()
    window.show()
    sys.exit(app.exec())


if __name__ == '__main__':
    main()

    QDockWidget

    import sys

from PyQt6.QtCore import *
from PyQt6.QtWidgets import *


class Demo(QMainWindow):
    def __init__(self, parent=None):
        super(Demo, self).__init__(parent)
        layout = QHBoxLayout()
        bar = self.menuBar()
        file = bar.addMenu('File')
        file.addAction('New')
        file.addAction('Save')
        file.addAction('quit')

        self.items = QDockWidget('Dockable', self)

        self.listWidget = QListWidget()
        self.listWidget.addItem('Item1')
        self.listWidget.addItem('Item2')
        self.listWidget.addItem('Item3')
        self.listWidget.addItem('Item4')

        self.items.setWidget(self.listWidget)
        self.items.setFloating(False)
        self.setCentralWidget(QTextEdit())
        self.addDockWidget(Qt.DockWidgetArea.BottomDockWidgetArea, self.items)

        self.setLayout(layout)
        self.setWindowTitle('Dock')


if __name__ == '__main__':
    app = QApplication(sys.argv)
    demo = Demo()
    demo.show()
    sys.exit(app.exec())
