def split_data(SOURCE, TRAINING, TESTING, SPLIT_SIZE):
  # [START] THIS IS YOUR CODE
  source_list = random.sample(os.listdir(SOURCE), len(os.listdir(SOURCE)))

  checked_list = []
  for im in source_list:
      if os.path.getsize(SOURCE + im) > 0:
          checked_list.append(im)

  train_list = checked_list[:int(len(checked_list) * SPLIT_SIZE)]
  test_list = checked_list[int(len(checked_list) * SPLIT_SIZE):]

  for tr in train_list:
      copyfile(SOURCE + tr, TRAINING + tr)

  for te in test_list:
      copyfile(SOURCE + te, TESTING + te)

  # [END] THIS IS YOUR CODE

def get_data(filename):
    with open(filename) as training_file:
      # [START] YOUR CODE HERE
        d = np.loadtxt(filename, delimiter=",", skiprows=1)
        labels = d[:, 0].astype(int).reshape(-1)
        images = d[:, 1:].astype(float).reshape(-1, 28, 28)
      # [END] YOUR CODE HERE
    return images, labels
