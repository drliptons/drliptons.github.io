    early_stopping(monitor="val_loss", silent_epochs=5)
    Arguments
    monitor (str) parameter to observe (Default: val_loss)
    silent_epochs (int) number of epochs without improvement before stop training (Default: 5)

    Reduce learning rate

                     min_learning_rate=0.001, 
                     silent_epochs=5, 
                     mode="min", 
                     verbose=1)
    Arguments
    monitor (str) parameter to observe (Default: val_loss)
    min_learning_rate (float) lower boundary of the learning rate (Default: 0.001)

    import tensorflow as tf


# Early stopping
def early_stopping(monitor="val_loss", silent_epochs=5):
    """
    Stop a training when a certain parameter (monitor) is not improving
    :param monitor: (str) parameter to observe (Default: val_loss)
    :param silent_epochs: (int) number of epochs without improvement before stop training (Default: 5)
    :return: tf.keras.callbacks.EarlyStopping()
    """
    early_stop = tf.keras.callbacks.EarlyStopping(patience=silent_epochs, monitor=monitor)
    return early_stop


# Reduce learning rate
def reduce_learning_rate(monitor="val_loss", min_learning_rate=0.001, silent_epochs=5, mode="min", verbose=1):
    """
    Call ReduceLROnPlateau() when an observed metrics has stopped improving
    :param monitor: (str) matrix to be monitored (Default: val_loss)
    :param min_learning_rate: (float) lower boundary of the learning rate (Default: 0.001)
    :param silent_epochs: (int) number of epochs without improvement before stop training (Default: 5)
    :param mode: one of {'auto', 'min', 'max'} min: reduce lr when 'monitor' is stopped decreasing, max: reduce lr when
    'monitor' has stopped increasing, auto: automatically inferred from the 'monitor'
    :param verbose: (int) 0: not showing any messages, 1: showing update messages
    :return: tf.keras.callbacks.ReduceLROnPlateau()
    """
    reduce_lr = tf.keras.callbacks.ReduceLROnPlateau(monitor=monitor,
                                                     min_lr=min_learning_rate,
                                                     patience=5,
                                                     mode="min",
                                                     verbose=verbose)
    return reduce_lr


# Create model checkpoint
def model_checkpoint(monitor="val_loss", model_path_in_h5="./model_best_result.h5"):
    """
    Create a model checkpoint when a model has improved in a certain parameter (monitor)
    :param monitor: parameter to observe (Default: val_loss)
    :param model_path_in_h5: to be saved in when reaching the observed value has improved
    (Default: ./model_best_result.h5)
    :return: tf.keras.callbacks.ModelCheckpoint()
    """
    checkpoint = tf.keras.callbacks.ModelCheckpoint(monitor=monitor,
                                                    filepath=model_path_in_h5,
                                                    save_best_only=True)
    return checkpoint


def callbacks_all():
    """
    Call all callbacks at once with suing their default values
    :return: list of callbacks
    """
    model_callbacks = [early_stopping(),
                       reduce_learning_rate(),
                       model_checkpoint()]

    return model_callbacks
